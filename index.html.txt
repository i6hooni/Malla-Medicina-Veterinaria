<script>
const ramos=document.querySelectorAll('.ramo');

function verificar(){
  ramos.forEach(r=>{
    const prereq=r.dataset.prereq;
    if(prereq){
      const ok=prereq.split(',').every(id=>{
        const p=document.getElementById(id.trim());
        return p&&p.classList.contains('aprobado');
      });
      ok? r.classList.remove('bloqueado') : r.classList.add('bloqueado');
    }
  });
}

ramos.forEach(r=>{
  r.addEventListener('click',()=>{
    if(r.classList.contains('bloqueado'))return;
    r.classList.toggle('aprobado');
    verificar();
  });
});

document.getElementById('reset').addEventListener('click',()=>{
  ramos.forEach(r=>r.classList.remove('aprobado'));
  verificar();
});

verificar(); // inicial
</script>
